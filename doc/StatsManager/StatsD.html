def self.count(stat, amount, uid=false, action=false)

def self.time(stat, uid=false, action=false, &block)
  start_t = Time.now
  yield block
  end_t = Time.now
  
  stat = STAT_PREFIX + stat
  client.timing(stat, ((end_t - start_t)*1000).round)
end
